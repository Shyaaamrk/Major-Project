# ===== Cell 4 — Preprocessing =====
from tensorflow.keras.preprocessing.image import ImageDataGenerator

# Define image size and batch size
img_size = 224
batch_size = 8

# Data augmentation & normalization for training
train_datagen = ImageDataGenerator(
    rescale=1./255,
    rotation_range=20,
    width_shift_range=0.1,
    height_shift_range=0.1,
    shear_range=0.1,
    zoom_range=0.1,
    horizontal_flip=True,
    # validation_split removed for tiny dataset
)

# Train generator (using all available training data)
train_generator = train_datagen.flow_from_directory(
    'varicose_dataset/train',
    target_size=(img_size, img_size),
    batch_size=batch_size,
    class_mode='binary',
    # subset removed as there's no split
)

# Test data generator (no augmentation)
test_datagen = ImageDataGenerator(rescale=1./255)
test_generator = test_datagen.flow_from_directory(
    'varicose_dataset/test',
    target_size=(img_size, img_size),
    batch_size=1,
    class_mode='binary',
    shuffle=False
)

print("\n✅ Data generators are ready!")
# ===== Cell 5 — Build CNN Model =====
from tensorflow.keras import layers, models

# Define the CNN architecture
model = models.Sequential([
    # 1️⃣ Convolutional Block 1
    layers.Conv2D(32, (3,3), activation='relu', input_shape=(img_size, img_size, 3)),
    layers.MaxPooling2D(2,2),

    # 2️⃣ Convolutional Block 2
    layers.Conv2D(64, (3,3), activation='relu'),
    layers.MaxPooling2D(2,2),

    # 3️⃣ Convolutional Block 3
    layers.Conv2D(128, (3,3), activation='relu'),
    layers.MaxPooling2D(2,2),

    # 4️⃣ Flatten + Fully Connected Layers
    layers.Flatten(),
    layers.Dense(128, activation='relu'),
    layers.Dropout(0.5),  # Prevent overfitting
    layers.Dense(1, activation='sigmoid')  # Binary output (0=Normal, 1=Varicose)
])

# Compile the model
model.compile(
    optimizer='adam',
    loss='binary_crossentropy',
    metrics=['accuracy']
)

# Summary
model.summary()
# ===== Cell 6 — Train the Model =====
import matplotlib.pyplot as plt

# Train for a few epochs (increase epochs if you have more data)
history = model.fit(
    train_generator,
    epochs=10,  # You can increase this to 20 or 25 for better results
    verbose=1
)

# Plot training and validation accuracy
plt.figure(figsize=(8,5))
plt.plot(history.history['accuracy'], label='Train Accuracy')
# There is no validation_accuracy in history because no validation_data was provided

plt.xlabel('Epoch')
plt.ylabel('Accuracy')

plt.show()

# Plot training and validation loss

plt.plot(history.history['loss'], label='Train Loss')

plt.title('Model Loss')
plt.xlabel('Epoch')
plt.ylabel('Loss')
plt.legend()
plt.show()

print("✅ Model training complete!")
# ===== Cell 7 — Evaluate Model on Validation Data =====
from sklearn.metrics import classification_report, confusion_matrix
import seaborn as sns
import numpy as np

# Get predictions using the test_generator
test_generator.reset()
predictions = model.predict(test_generator)
predicted_classes = (predictions > 0.5).astype("int32").flatten()

# True labels from test_generator
true_classes = test_generator.classes
class_labels = list(test_generator.class_indices.keys())

# Confusion matrix
cm = confusion_matrix(true_classes, predicted_classes)

plt.figure(figsize=(6,5))
sns.heatmap(cm, annot=True, fmt='d', cmap='Blues', xticklabels=class_labels, yticklabels=class_labels)
plt.xlabel('Predicted Label')
plt.ylabel('True Label')
plt.title('Confusion Matrix')
plt.show()

# Classification report
report = classification_report(true_classes, predicted_classes, target_names=class_labels)
print("📊 Classification Report:\n", report)

# Accuracy
accuracy = np.mean(predicted_classes == true_classes)
print(f"✅ Overall Model Accuracy: {accuracy*100:.2f}%")

from google.colab import files
from tensorflow.keras.preprocessing import image
import numpy as np
import matplotlib.pyplot as plt
import cv2

# Upload an image
uploaded = files.upload()

for filename in uploaded.keys():
    # Load and preprocess the image
    img_path = filename
    img = image.load_img(img_path, target_size=(img_size, img_size))
    img_array = image.img_to_array(img)
    img_array = np.expand_dims(img_array, axis=0) / 255.0  # Normalize

    # Make prediction
    prediction = model.predict(img_array)
    result = "🩺 Varicose Vein Detected" if prediction[0][0] > 0.5 else "✅ Normal Vein"

    # Confidence
    confidence = prediction[0][0] if prediction[0][0] > 0.5 else (1 - prediction[0][0])

    # Display result
    plt.imshow(plt.imread(img_path))
    plt.axis('off')
    plt.title(f"{result}\nConfidence: {confidence*100:.2f}%")
    plt.show()

    print(f"Prediction: {result}")
    print(f"Confidence Score: {confidence*100:.2f}%")
